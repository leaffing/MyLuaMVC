 
-- LuaMVC中 lua部分的启动脚本

require('NotificationType') 
require('ViewNames')

UnityEngine = CS.UnityEngine
GameObject = CS.UnityEngine.GameObject
LuaMVC = CS.LuaMVC.LuaApplicationFacade -- 用于给luaMVC发送通知
AssetLoader = CS.LuaMVC.AssetLoader  -- 用于加载Resources/Assetbundle资源
 
function awake()
	print('lua part framework start up.')
	self:RegisterLuaCommand("StartUpCommand") 
	self:RegisterLuaCommand("AddTestDataCommand") 

	self:RegisterLuaProxy("TestProxy")

	local testView = GameObject.Find("TestView")
	testView:AddComponent(typeof(CS.LuaMVC.LuaMonobehaviour)):Init('TestView') 
	self:RegisterLuaMediator('TestViewMediator')

--[[

function ondestroy()
	print('lua part framework shut down.')
