-- 测试视图

TestView = {} 

TestView.ViewName = ViewNames.TestView
local transform
  
TestView.awake = function()  
    transform = TestView.self.transform
    TestView.button = transform:GetComponentInChildren(typeof(CS.UnityEngine.UI.Button))
end

TestView.Open = function()
    TestView.panel:SetActive(true)
end

TestView.Close = function()
    TestView.panel:SetActive(false)
end

return TestView
